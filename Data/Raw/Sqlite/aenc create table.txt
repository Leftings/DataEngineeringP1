


CREATE TABLE product
(
    ID INT,
    NAME TEXT,
    DESCRIPTION TEXT,
    PROD_SIZE TEXT,
    COLOR TEXT,
    QUANTITY INT,
    UNIT_PRICE INT,
    PICTURE_NAME TEXT,
    CATEGORY TEXT,

    PRIMARY KEY (ID)
)

CREATE TABLE state
(
    STATE_ID VARCHAR(2),
    STATE_NAME TEXT,
    STATE_CAPITAL TEXT,
    COUNTRY TEXT,
    REGION TEXT,

    PRIMARY KEY (STATE_ID)
)

CREATE TABLE department
(
    DEPT_ID INT,
    DEPT_NAME TEXT,
    DEPT_HEAD_ID INT,

    PRIMARY KEY (DEPT_ID)
)

CREATE TABLE employee
(
    EMP_ID INT,
    MANAGER_ID INT,
    EMP_FNAME TEXT,
    EMP_LNAME TEXT,
    DEPT_ID INT,
    STREET TEXT,
    CITY TEXT,
    STATE VARCHAR(2),
    ZIP_CODE INT,
    PHONE INT,
    STATUS TEXT,
    SS_NUMBER INT,
    SALARY INT,
    START_DATE TEXT,
    BIRTH_DATE TEXT,
    BENE_HEALT_INS TEXT,
    BENE_LIFE_INS TEXT,
    BENE_DAY_CARE TEXT,
    SEX TEXT,

    PRIMARY KEY (EMP_ID),
    FOREIGN KEY (DEPT_ID) REFERENCES department(DEPT_ID),
    FOREIGN KEY (STATE) REFERENCES state(STATE_ID)
)

CREATE TABLE customer
(
    CUSTOMER_ID INT,
    FNAME TEXT,
    LNAME TEXT,
    ADDRESS TEXT,
    CITY TEXT,
    STATE VARCHAR(2),
    ZIP TEXT,
    PHONE INT,
    COMPANY_NAME TEXT,

    PRIMARY KEY (CUSTOMER_ID),
    FOREIGN KEY (STATE) REFERENCES state(STATE_ID)
)

CREATE TABLE bonus
(
    EMP_ID INT,
    BONUS_DATE TEXT,
    BONUS_AMOUNT INT,

    FOREIGN KEY (EMP_ID) REFERENCES employee(EMP_ID)
)

CREATE TABLE sales_order
(
    ID INT,
    CUSTOMER_ID INT,
    ORDER_DATE TEXT,
    REGION TEXT,
    SALES_REP INT,

    PRIMARY KEY (ID),
    FOREIGN KEY (CUSTOMER_ID) REFERENCES customer(CUSTOMER_ID),
    FOREIGN KEY (SALES_REP) REFERENCES employee(EMP_ID)
)

CREATE TABLE sales_order_item
(
    ID INT,
    LINE_ID INT,
    QUANTITY INT,
    SHIP_DATE TEXT,

    FOREIGN KEY (ID) REFERENCES sales_order(ID)
)
